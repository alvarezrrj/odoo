# openupgrade_migrate.py

def migrate_currency_rate_handling(env):
    """Migrate currency rate handling logic"""
    # The l10n_ar_currency_rate field has been removed
    # Currency rates are now handled by standard Odoo currency system
    
    # For invoices that had custom rates, we may need to update related records
    cr = env.cr
    
    if openupgrade.column_exists(cr, 'account_move', 'openupgrade_legacy_l10n_ar_currency_rate'):
        cr.execute("""
            SELECT id, openupgrade_legacy_l10n_ar_currency_rate, currency_id, company_id, invoice_date
            FROM account_move 
            WHERE openupgrade_legacy_l10n_ar_currency_rate IS NOT NULL 
            AND openupgrade_legacy_l10n_ar_currency_rate != 1.0
        """)
        
        for move_id, legacy_rate, currency_id, company_id, invoice_date in cr.fetchall():
            # Log moves that had custom rates for manual review
            env['ir.logging'].create({
                'name': 'l10n_ar_migration',
                'type': 'server',
                'level': 'INFO',
                'message': f'Move {move_id} had custom currency rate {legacy_rate}. '
                          f'Please verify currency conversion is correct.',
                'path': 'l10n_ar_migration',
                'line': '1',
                'func': 'migrate_currency_rate_handling'
            })

def migrate_partner_document_types(env):
    """Handle removal of l10n_ar_special_purchase_document_type_ids"""
    cr = env.cr
    
    if openupgrade.table_exists(cr, 'openupgrade_legacy_partner_doc_types'):
        cr.execute("SELECT partner_id, document_type_ids FROM openupgrade_legacy_partner_doc_types")
        
        for partner_id, doc_type_ids in cr.fetchall():
            # Log partners that had special document types for manual review
            env['ir.logging'].create({
                'name': 'l10n_ar_migration',
                'type': 'server', 
                'level': 'WARNING',
                'message': f'Partner {partner_id} had special purchase document types: {doc_type_ids}. '
                          f'This field has been removed. Please review partner configuration.',
                'path': 'l10n_ar_migration',
                'line': '1',
                'func': 'migrate_partner_document_types'
            })

def update_tax_computation_data(env):
    """Update any data related to tax computation changes"""
    # The _l10n_ar_prices_and_taxes method has been completely refactored
    # Generally this should not require data migration, but we should verify
    # that tax calculations remain consistent
    
    # Run a validation check on recent invoices
    recent_invoices = env['account.move'].search([
        ('company_id.account_fiscal_country_id.code', '=', 'AR'),
        ('state', '=', 'posted'),
        ('invoice_date', '>=', '2024-01-01')  # Adjust date as needed
    ], limit=100)
    
    for invoice in recent_invoices:
        try:
            # Trigger recomputation to ensure consistency
            invoice.invoice_line_ids._compute_all_tax()
        except Exception as e:
            env['ir.logging'].create({
                'name': 'l10n_ar_migration',
                'type': 'server',
                'level': 'ERROR', 
                'message': f'Tax recomputation failed for invoice {invoice.id}: {str(e)}',
                'path': 'l10n_ar_migration',
                'line': '1',
                'func': 'update_tax_computation_data'
            })

@openupgrade.migrate()
def migrate(env, version):
    migrate_currency_rate_handling(env)
    migrate_partner_document_types(env)

def validate_cuit_numbers(env):
    """Validate CUIT numbers with enhanced validation"""
    partners_with_cuit = env['res.partner'].search([
        ('l10n_latam_identification_type_id.l10n_ar_afip_code', '=', '80'),  # CUIT
        ('vat', '!=', False)
    ])
    
    invalid_cuits = []
    for partner in partners_with_cuit:
        try:
            partner._check_vat()
        except Exception as e:
            invalid_cuits.append((partner.id, partner.vat, str(e)))
    
    if invalid_cuits:
        env['ir.logging'].create({
            'name': 'l10n_ar_migration_validation',
            'type': 'server',
            'level': 'WARNING',
